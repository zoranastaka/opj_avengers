// funkcija za prikaz grešaka pri prijavi

function login_try()
{
    if(!isset($_POST['email']) || !isset($_POST['lozinka']) || empty($_POST['email']) || empty($_POST['lozinka']))
    {
        echo 'Unesite e-mail i lozinku';
    }else
    {
        if(!isset($_COOKIE['user_id']))
        {
            echo 'Pogrešan e-mail\lozinka';
        };
    };
};
