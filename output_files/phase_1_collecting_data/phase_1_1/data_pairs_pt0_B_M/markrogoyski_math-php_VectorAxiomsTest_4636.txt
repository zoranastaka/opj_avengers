    /**
     * @test Axiom: A⋅A⊥ = -A⊥⋅A
     * Swapping operands changes the sign of the perp dot product
     * @dataProvider dataProviderForPerpendicularIdentity
     */

    public function testPerpDotProdcutSwapOperandsChangeSign(array $A)
    {
        // Given
        $A  = new Vector($A);
        $A⊥ = $A->perpendicular();

        // When
        $A⋅A⊥ = $A->dotProduct($A⊥);
        $A⊥⋅A = $A⊥->dotProduct($A);

        // Then
        $this->assertEquals($A⋅A⊥, -$A⊥⋅A);
    }
