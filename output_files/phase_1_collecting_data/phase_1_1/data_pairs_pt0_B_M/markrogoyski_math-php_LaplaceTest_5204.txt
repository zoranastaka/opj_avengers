    /**
     * @test         cdf
     * @dataProvider dataProviderForCdf
     * @param        float $x
     * @param        float $μ
     * @param        float $b
     * @param        float $expected_cdf
     */

    public function testCdf(float $x, float $μ, float $b, float $expected_cdf)
    {
        // Given
        $laplace = new Laplace($μ, $b);

        // When
        $cdf = $laplace->cdf($x);

        // Then
        $this->assertEqualsWithDelta($expected_cdf, $cdf, 0.000001);
    }
