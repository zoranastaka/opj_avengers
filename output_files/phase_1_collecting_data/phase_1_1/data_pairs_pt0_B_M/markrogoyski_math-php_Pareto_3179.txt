    /**
     * Mean of the distribution
     *
     * μ = ∞ for a ≤ 1
     *
     *      ab
     * μ = ----- for a > 1
     *     a - 1
     *
     * @return float
     */

    public function mean(): float
    {
        $a = $this->a;
        $b = $this->b;

        if ($a <= 1) {
            return \INF;
        }

        return $a * $b / ($a - 1);
    }
