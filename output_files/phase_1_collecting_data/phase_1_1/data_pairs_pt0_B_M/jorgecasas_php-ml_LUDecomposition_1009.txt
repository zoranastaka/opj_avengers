    /**
     * Is the matrix nonsingular?
     *
     * @return bool true if U, and hence A, is nonsingular.
     */

    public function isNonsingular(): bool
    {
        for ($j = 0; $j < $this->n; ++$j) {
            if ($this->LU[$j][$j] == 0) {
                return false;
            }
        }

        return true;
    }
