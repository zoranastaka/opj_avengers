    /**
     * draw a clipping rectangle, all the elements added after this will be clipped
     *
     * @param $x1
     * @param $y1
     * @param $width
     * @param $height
     */

    function clippingRectangle($x1, $y1, $width, $height)
    {
        $this->save();
        $this->addContent(sprintf("\n%.3F %.3F %.3F %.3F re W n", $x1, $y1, $width, $height));
    }
