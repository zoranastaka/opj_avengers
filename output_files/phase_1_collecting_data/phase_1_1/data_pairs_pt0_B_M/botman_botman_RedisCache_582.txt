    /**
     * Determine if an item exists in the cache.
     *
     * @param  string $key
     * @return bool
     */

    public function has($key)
    {
        /*
         * Version >= 4.0 of phpredis returns an integer instead of bool
         */
        $check = $this->redis->exists($this->decorateKey($key));

        if (is_bool($check)) {
            return $check;
        }

        return $check > 0;
    }
