    /**
     * Implementation of \ArrayAccess::offsetSet()
     *
     * @param string $offset
     * @param mixed $value
     * @return void
     * @link http://www.php.net/manual/en/arrayaccess.offsetset.php
     */

    public function offsetSet($offset, $value)
    {
        $this->_data[$offset] = $value;
    }
