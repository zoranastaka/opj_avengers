    /**
     * The dispersion
     * Dv = (Σv² / N) - (Σv / N)²
     *
     * @param string $v 'x' or 'y'
     *
     * @return number
     */

    private function dispersion($v) {
        return (($this->sumSquared($this->{$v.'Vector'}) / $this->count) - 
        $this->square($this->sum($this->{$v.'Vector'}) / $this->count));
    }
