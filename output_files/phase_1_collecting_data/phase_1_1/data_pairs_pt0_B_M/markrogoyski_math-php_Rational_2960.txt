    /**
     * Divide by an integer
     *
     * @param int $int
     *
     * @return Rational
     */

    private function divideInt(int $int): Rational
    {
        $w = $this->whole;
        $n = $this->numerator;
        $d = $this->denominator;
        return new Rational(0, $w * $d + $n, $int * $d);
    }
