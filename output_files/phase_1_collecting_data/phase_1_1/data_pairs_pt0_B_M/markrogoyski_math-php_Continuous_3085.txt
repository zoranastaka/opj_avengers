    /**
     * CDF above - Probability of being above x to âˆž
     * Area under a continuous distribution, that lies above a specified point.
     *
     * P(above) = 1 - CDF(x)
     *
     * @param float $x
     *
     * @return float
     */

    public function above(float $x): float
    {
        return 1 - $this->cdf($x);
    }
