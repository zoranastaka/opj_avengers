    /**
     * Convert the numerator to superscript character
     *
     * @return string
     */

    private function numeratorToSuperscript(): string
    {
        return $this->toSuperOrSubscript(
            \abs($this->numerator),
            ['⁰', '¹', '²', '³', '⁴', '⁵', '⁶', '⁷', '⁸', '⁹']
        );
    }
