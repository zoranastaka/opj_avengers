    /**
     * Mode of the distribution
     *
     * 0    for p < ½
     * 0,1  for p = ½
     * 1    for p > ½
     *
     * @return float[]
     */

    public function mode(): array
    {
        $p = $this->p;
        $½ = 0.5;

        if ($p < $½) {
            return [0];
        }
        if ($p == $½) {
            return [0, 1];
        }
        return [1];
    }
