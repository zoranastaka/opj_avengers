    /**
     * Map raise to a power
     *
     * @param  array  $xs
     * @param  number $n
     *
     * @return array
     */

    public static function pow(array $xs, $n): array
    {
        return \array_map(
            function ($x) use ($n) {
                return $x ** $n;
            },
            $xs
        );
    }
