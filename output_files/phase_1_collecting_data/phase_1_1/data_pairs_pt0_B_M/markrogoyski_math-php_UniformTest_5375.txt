    /**
     * @test         mode
     * @dataProvider dataProviderForMean
     * @param        float $a
     * @param        float $b
     * @throws       \Exception
     */

    public function testMode(float $a, float $b)
    {
        // Given
        $uniform = new Uniform($a, $b);

        // When
        $mode = $uniform->mode();

        // Then
        $this->assertGreaterThanOrEqual($a, $mode);
        $this->assertLessThanOrEqual($b, $mode);
    }
