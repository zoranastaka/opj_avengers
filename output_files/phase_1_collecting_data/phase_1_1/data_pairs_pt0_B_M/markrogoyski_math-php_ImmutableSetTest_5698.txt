    /**
     * @test add does nothing
     */

    public function testAddDoesNothing()
    {
        // Given
        $A = new ImmutableSet([1, 2, 3, 4]);
        $B = $A->copy();

        // When
        $A->add(5);
        $A->add(6);
        $A->add([7, 8, 9]);
        $A->add(new Set(['a', 'b']));

        // Then
        $this->assertEquals($B, $A);
        $this->assertEquals($B->asArray(), $A->asArray());
    }
