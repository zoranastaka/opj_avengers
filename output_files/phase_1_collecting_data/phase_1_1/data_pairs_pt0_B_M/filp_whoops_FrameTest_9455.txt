    /**
     * @covers Whoops\Exception\Frame::getFileContents
     * @testWith ["[internal]"]
     *           ["Unknown"]
     * @see https://github.com/filp/whoops/pull/599
     */

    public function testGetFileContentsWhenFrameIsNotRelatedToSpecificFile($fakeFilename)
    {
