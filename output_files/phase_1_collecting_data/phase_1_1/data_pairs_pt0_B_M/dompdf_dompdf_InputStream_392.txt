    /**
     * Get all characters until EOF.
     * @note This performs bounds checking
     *
     * @return string|bool
     */

    public function remainingChars() {
        if ($this->char < $this->EOF) {
            $data = substr($this->data, $this->char);
            $this->char = $this->EOF;
            return $data;
        } else {
            return false;
        }
