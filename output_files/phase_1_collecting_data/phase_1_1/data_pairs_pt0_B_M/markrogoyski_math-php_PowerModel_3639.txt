    /**
     * Get regression parameters (a and b)
     *
     * @param array $params
     *
     * @return array [ a => number, b => number ]
     */

    public function getModelParameters(array $params): array
    {
        return [
            'a' => $params[self::$A],
            'b' => $params[self::$B],
        ];
    }
