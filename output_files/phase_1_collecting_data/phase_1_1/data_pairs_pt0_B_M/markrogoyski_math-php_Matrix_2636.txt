    /**
     * Is this matrix the same size and type as some other matrix?
     *
     * @param Matrix $B
     *
     * @return bool
     */

    protected function isEqualSizeAndType(Matrix $B): bool
    {
        if ($this->getObjectType() !== $B->getObjectType()) {
            return false;
        }

        $m = $this->m;
        $n = $this->n;

        // Same dimensions
        if ($m != $B->m || $n != $B->n) {
            return false;
        }

        return true;
    }
