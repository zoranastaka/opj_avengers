    /**
     * @covers Whoops\Exception\FrameCollection::getArray

    public function testGetArrayImmutable()

        $frames = $this->getFrameCollectionInstance();
        $arr = $frames->getArray();
        $arr[0] = 'foobar';

        $this->assertNotSame($arr[0], $newCopy);
