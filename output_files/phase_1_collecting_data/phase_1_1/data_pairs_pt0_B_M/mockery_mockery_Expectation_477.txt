    /**
     * Alias to andSet(). Allows the natural English construct
     * - set('foo', 'bar')->andReturn('bar')
     *
     * @param string $name
     * @param mixed $value
     * @return self
     */

    public function set($name, $value)
    {
        return call_user_func_array(array($this, 'andSet'), func_get_args());
    }
