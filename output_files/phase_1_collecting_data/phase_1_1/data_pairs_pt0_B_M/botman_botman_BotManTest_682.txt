    /** @test */

    public function it_can_reply_a_random_message()
    {
        $randomMessages = ['foo', 'bar', 'baz'];
        $botman = $this->getBot([]);
        $driver = m::mock(FakeDriver::class)->makePartial();
        $botman->setDriver($driver);
        $botman->randomReply($randomMessages, []);
