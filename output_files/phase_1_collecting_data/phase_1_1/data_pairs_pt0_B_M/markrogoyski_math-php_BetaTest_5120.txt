    /**
     * @test rand
     */

    public function testRand()
    {
        foreach (\range(1, 10) as $α) {
            foreach (\range(1, 10) as $β) {
                // Given
                $beta = new Beta($α, $β);
                foreach (\range(1, 3) as $_) {
                    // When
                    $random = $beta->rand();

                    // Then
                    $this->assertTrue(\is_numeric($random));
                }
            }
        }
    }
