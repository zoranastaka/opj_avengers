    /**
     * Raw data with each observation labeled
     * ['B1' => ['Heating value' => 18373, 'C' => 41.61500, 'H' => 6.565000, ... ]]
     *
     * @return number[]
     */

    public function getLabeledYData(): array
    {
        return \array_map(
            function (array $data) {
                return \array_combine(self::Y_LABELS, $data);
            },
            self::Y_DATA
        );
    }
