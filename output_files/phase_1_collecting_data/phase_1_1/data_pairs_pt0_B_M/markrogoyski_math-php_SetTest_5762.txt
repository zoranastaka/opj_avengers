    /**
     * @test         asArray
     * @dataProvider dataProviderForSingleSet
     * @param        array $members
     */

    public function testAsArrayAnotherWay(array $members)
    {
        // Given
        $set   = new Set($members);

        // When
        $array = $set->asArray();
        $new_set = new Set($array);

        // Then
        $this->assertEquals($new_set, $set);
    }
