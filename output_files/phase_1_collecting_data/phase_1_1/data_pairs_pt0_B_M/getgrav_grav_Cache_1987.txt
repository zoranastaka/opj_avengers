    /**
     * is this driver a volatile driver in that it resides in PHP process memory
     *
     * @param string $setting
     * @return bool
     */

    public function isVolatileDriver($setting)
    {
        if (in_array($setting, ['apc', 'apcu', 'xcache', 'wincache'])) {
            return true;
        }

        return false;
    }
