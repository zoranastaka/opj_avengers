    /**
     * @test         variance

     * @param        float $a
     * @param        float $b
     * @param        float $expected
     * @throws       \Exception
     */

    public function testVariance(float $a, float $b, float $expected)

        $uniform = new Uniform($a, $b);

        $this->assertEqualsWithDelta($expected, $variance, 0.00001);
