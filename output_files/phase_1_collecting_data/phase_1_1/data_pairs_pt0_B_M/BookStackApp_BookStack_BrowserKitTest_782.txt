    /**
     * Assert that the current page matches a given URI.
     *
     * @param  string  $uri
     * @return $this
     */

    protected function seePageUrlIs($uri)
    {
        $this->assertEquals(
            $uri, $this->currentUri, "Did not land on expected page [{$uri}].\n"
        );

        return $this;
    }
