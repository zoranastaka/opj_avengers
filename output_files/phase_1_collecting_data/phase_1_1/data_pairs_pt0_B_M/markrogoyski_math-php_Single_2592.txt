    /**
     * Map Divide
     *
     * @param  array $xs
     * @param  number $k Number to divide each element by
     *
     * @return array
     */

    public static function divide(array $xs, $k): array
    {
        return \array_map(
            function ($x) use ($k) {
                return $x / $k;
            },
            $xs
        );
    }
