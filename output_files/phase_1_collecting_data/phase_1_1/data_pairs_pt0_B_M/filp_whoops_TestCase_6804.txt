    /**
     * @param  string $a
     * @param  string $b
     * @return void
     */

    protected function assertStringContains($a, $b)
    {
        if (method_exists($this, 'assertStringContainsString')) {
            $this->assertStringContainsString($a, $b);
        } else {
            $this->assertContains($a, $b);
        }
    }
