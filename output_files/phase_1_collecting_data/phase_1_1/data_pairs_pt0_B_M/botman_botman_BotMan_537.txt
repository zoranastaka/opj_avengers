    /**
     * Listen for messaging service events.
     *
     * @param array|string $names
     * @param Closure|string $callback
     */

    public function on($names, $callback)
    {
        if (! is_array($names)) {
            $names = [$names];
        }

        $callable = $this->getCallable($callback);

        foreach ($names as $name) {
            $this->events[] = [
                'name' => $name,
                'callback' => $callable,
            ];
        }
    }
