    /**
     * Creates the power set of A.
     *
     * @return Set[]
     */

    public static function power(self $a): array
    {
        $power = [new self()];

        foreach ($a as $multiplicand) {
            foreach ($power as $multiplier) {
                $power[] = new self(array_merge([$multiplicand], $multiplier->toArray()));
            }
        }

        return $power;
    }
