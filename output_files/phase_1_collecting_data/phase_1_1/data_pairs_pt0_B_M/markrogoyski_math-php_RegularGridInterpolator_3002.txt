    /**
     * @param array $indices
     * @param array $normDistances
     *
     * @return float|int
     */

    private function evaluateNearest(array $indices, array $normDistances)
    {
        $idxRes = [];
        foreach (Iter::zip($indices, $normDistances) as [$i, $yi]) {
            $idxRes[] = $yi <= 0.5
                ? $i
                : $i + 1;
        }

        return $this->flatCall($this->values, $idxRes);
    }
