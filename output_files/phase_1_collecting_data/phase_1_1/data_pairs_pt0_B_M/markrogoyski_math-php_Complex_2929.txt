    /**
     * Polar form
     * https://en.wikipedia.org/wiki/Complex_number#Polar_form
     *
     * z = a + bi = r(cos(θ) + i  sin(θ))
     * Where
     *  r = |z|
     *  θ = arg(z) (in radians)
     *
     * @return Complex
     */

    public function polarForm(): Complex
    {
        $r = $this->abs();
        $θ = $this->arg();

        return new Complex($r * \cos($θ), $r * \sin($θ));
    }
