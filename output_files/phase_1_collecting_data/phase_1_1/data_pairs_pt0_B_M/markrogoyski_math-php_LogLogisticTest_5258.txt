    /**
     * @test         median
     * @dataProvider dataProviderForMean
     * @param        float $α
     * @param        float $β
     */

    {

        $logLogistic = new LogLogistic($α, $β);

        $this->assertEqualsWithDelta($α, $median, 0.00001);
