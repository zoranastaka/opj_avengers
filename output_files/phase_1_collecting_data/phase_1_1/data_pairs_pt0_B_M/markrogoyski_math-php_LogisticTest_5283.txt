    /**
     * @test         cdf
     * @dataProvider dataProviderForCdf
     * @param        float $x
     * @param        float $μ
     * @param        float $s
     * @param        float $expected_cdf
     */

    public function testCdf(float $x, float $μ, float $s, float $expected_cdf)
    {
        // Given
        $logistic = new Logistic($μ, $s);

        // When
        $cdf = $logistic->cdf($x);

        // Then
        $this->assertEqualsWithDelta($expected_cdf, $cdf, 0.000001);
    }
