    /**
     * Map subtract
     *
     * @param  array $xs
     * @param  number $k Number to subtract from each element
     *
     * @return array
     */

    public static function subtract(array $xs, $k): array
    {
        return \array_map(
            function ($x) use ($k) {
                return $x - $k;
            },
            $xs
        );
    }
