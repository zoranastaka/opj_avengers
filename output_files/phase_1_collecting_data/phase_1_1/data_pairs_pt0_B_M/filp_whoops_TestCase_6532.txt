    /**
     * @param  string $a
     * @param  string $b
     * @return void
     */

    protected function assertStringNotContains($a, $b)
    {
        if (method_exists($this, 'assertStringNotContainsString')) {
            $this->assertStringNotContainsString($a, $b);
        } else {
            $this->assertNotContains($a, $b);
        }
    }
