    /**
     * Probability mass function
     *
     * q = (1 - p)  for k = 0
     * p            for k = 1
     *
     * @param  int $k number of successes  k âˆˆ {0, 1}
     *
     * @return float
     */

    public function pmf(int $k): float
    {
        Support::checkLimits(self::SUPPORT_LIMITS, ['k' => $k]);

        if ($k === 0) {
            return $this->q;
        } else {
            return $this->p;
        }
    }
