    /**
     * Map min value
     * Each element in array is compared against the value,
     * and the min of each is returned.
     *
     * @param  array  $xs
     * @param  number $value
     *
     * @return array
     */

    public static function min(array $xs, $value): array
    {
        return \array_map(
            function ($x) use ($value) {
                return \min($x, $value);
            },
            $xs
        );
    }
