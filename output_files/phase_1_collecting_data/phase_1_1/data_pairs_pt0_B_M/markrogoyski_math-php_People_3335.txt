    /**
     * Data for a person, with labels
     * ['height' => 198, 'weight' => 92, 'hairLength' => -1, ... ]
     *
     * @param string $name
     *
     * @return int[]
     */

    public function getPersonData(string $name): array
    {
        return \array_combine(self::LABELS, self::DATA[$name]);
    }
