    /**
     * Execute the console command.
     *
     * @return mixed
     */

    public function handle()
    {
        $type = strtolower($this->option('type'));

        switch ($type) {
            case 'referral':
                $this->referralPayouts();
                break;
            case 'reseller':
                $this->resellerPayouts();
                break;
        }
    }
