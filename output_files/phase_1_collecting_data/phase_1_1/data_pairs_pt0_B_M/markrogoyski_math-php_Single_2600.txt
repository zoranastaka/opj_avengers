    /**
     * Map max value
     * Each element in the array is compared against the value,
     * and the max of each is returned.
     *
     * @param  array  $xs
     * @param  number $value
     *
     * @return array
     */

    public static function max(array $xs, $value): array
    {
        return \array_map(
            function ($x) use ($value) {
                return \max($x, $value);
            },
            $xs
        );
    }
