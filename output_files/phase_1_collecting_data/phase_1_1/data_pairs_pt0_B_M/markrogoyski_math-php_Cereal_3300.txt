    /**
     * Raw data with each observation labeled
     * ['B1' => ['X1126.0' => 0.002682755, 'X1134.0' => 0.003370673, 'X1142.0' => 0.004085942, ... ]]
     *
     * @return number[]
     */

    public function getLabeledXData(): array
    {
        return \array_map(
            function (array $data) {
                return \array_combine(self::X_LABELS, $data);
            },
            self::X_DATA
        );
    }
