     * @test     mean

    public function testMean()
    {
        foreach (\range(-3, 3) as $μ) {
            foreach (\range(1, 3) as $s) {
                // Given
                $logistic = new Logistic($μ, $s);

                // When
                $mean = $logistic->mean();

                // Then
                $this->assertEquals($μ, $mean);
            }
        }
