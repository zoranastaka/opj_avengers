    /**
     * Default kind (if supported, based on current interpreter's version).
     *
     * @return string|null
     */

    public function getDefaultKind()
    {
        if ($this->hasKindsSupport()) {
            return \version_compare(\PHP_VERSION, '7.0', '>=') ? static::ONLY_PHP7 : static::ONLY_PHP5;
        }
    }
