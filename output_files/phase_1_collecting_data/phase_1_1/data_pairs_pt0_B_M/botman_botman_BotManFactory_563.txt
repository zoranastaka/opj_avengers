    /**
     * @param string $name
     * @param array $arguments
     * @return mixed
     */

    public static function __callStatic($name, $arguments)
    {
        try {
            return \call_user_func_array(self::$extensions[$name], $arguments);
        } catch (\Exception $e) {
            throw new \BadMethodCallException("Method [$name] does not exist.");
        }
    }
