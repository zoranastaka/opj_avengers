  /**
   * Returns true only if the array is associative.
   * @param array $array
   * @return bool True if the array is associative.
   */

  protected function isAssociativeArray($array)
  {
    if (!is_array($array)) {
      return false;
    }
    $keys = array_keys($array);
    foreach ($keys as $key) {
      if (is_string($key)) {
        return true;
      }
    }
    return false;
  }
