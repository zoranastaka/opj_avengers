    /** @test */

    public function it_can_retrieve_the_user()
    {
        $called = false;

        $botman = $this->getBot([
            'sender' => 'UX12345',
            'recipient' => 'general',
            'message' => 'Foo',
        ]);

        $botman->hears('foo', function ($bot) use (&$called) {
            $called = true;
            $this->assertSame('UX12345', $bot->getMessage()->getSender());
        });
        $botman->listen();
        $this->assertTrue($called);
    }
