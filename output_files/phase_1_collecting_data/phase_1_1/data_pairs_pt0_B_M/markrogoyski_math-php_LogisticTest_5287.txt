    /**
     * @test     median
     */

    public function testMedian()
    {
        foreach (\range(-3, 3) as $μ) {
            foreach (\range(1, 3) as $s) {
                // Given
                $logistic = new Logistic($μ, $s);

                // When
                $median = $logistic->median();

                // Then
                $this->assertEquals($μ, $median);
            }
        }
    }
