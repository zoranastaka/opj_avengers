    /**
     * Variance of the distribution
     *
     *        K (N - K) N - n
     * ÏƒÂ² = n - ------- -----
     *        N    N    N - 1
     *
     * @return float
     */

    public function variance(): float
    {
        $N = $this->N;
        $K = $this->K;
        $n = $this->n;

        return $n * ($K / $N) * (($N - $K) / $N) * (($N - $n) / ($N - 1));
    }
