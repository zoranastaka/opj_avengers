function lista_zadaci($to_do_id)
{
    global $zadaci;

    $svi = $GLOBALS['con']->prepare(' select count(status) from zadaci where lista_id = :lista_id ');
    $svi->execute([
        ':lista_id' => $to_do_id
    ]);
    $zadaci = $svi->fetchColumn();

    return $zadaci;
}
