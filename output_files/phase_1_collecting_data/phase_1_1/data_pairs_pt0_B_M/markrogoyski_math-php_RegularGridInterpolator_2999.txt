    /**
     * Count dimensions of a multi-dimensional array
     *
     * @param  array $array
     *
     * @return int
     */

    private function countDimensions(array $array): int
    {
        if (\is_array(\reset($array))) {
            $return = $this->countDimensions(\reset($array)) + 1;
        } else {
            $return = 1;
        }

        return $return;
    }
