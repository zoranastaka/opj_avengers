    /**
     * Throws an exception if the expectation has been configured to do so
     *
     * @throws \Throwable
     * @return void
     */

    private function throwAsNecessary($return)
    {
        if (!$this->_throw) {
            return;
        }

        if ($return instanceof \Throwable) {
            throw $return;
        }

        return;
    }
