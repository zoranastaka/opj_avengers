    /**
     * Lexical catchable patterns.

     * @return string[]
     */

    protected function getCatchablePatterns()
    {

            '[a-zA-Z]+[46]?', //ASCII and domain literal

            '[0-9]+',

            '::',
            '\s+?',
            '.',
            );
    }
