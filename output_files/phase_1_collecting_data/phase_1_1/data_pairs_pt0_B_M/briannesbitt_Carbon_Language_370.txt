    /**
     * Returns the variant such as Cyrillic/Latin.
     *
     * @return string|null
     */

    public function getVariantName(): ?string
    {
        if ($this->variant === 'Latn') {
            return 'Latin';
        }

        if ($this->variant === 'Cyrl') {
            return 'Cyrillic';
        }

        return $this->variant;
    }
