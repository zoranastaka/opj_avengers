    /**
     * @test         isNotMember
     * @dataProvider dataProviderForSingleSet
     * @param        array $members

    public function testIsNotMember(array $members)
    {

        $set = new Set($members);

        $this->assertTrue($set->isNotMember('TotallNotAMember'));
