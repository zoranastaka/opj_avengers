    /**
     * @test         pdf
     * @dataProvider dataProviderForPdf
     * @param        float $x   x ∈ (0,1)
     * @param        float $k   shape parameter α > 0
     * @param        float $θ   scale parameter θ > 0
     * @param        float $expectedPdf
     */

    public function testPdf(float $x, float $k, float $θ, float $expectedPdf)
    {
        // Given
        $gamma = new Gamma($k, $θ);

        // When

    }
