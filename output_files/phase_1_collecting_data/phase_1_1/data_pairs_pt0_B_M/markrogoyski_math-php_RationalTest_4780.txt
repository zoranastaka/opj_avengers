    /**
     * @test         toFloat returns the correct floating point number
     * @dataProvider dataProviderForToFloat
     * @param        number $w
     * @param        number $n
     * @param        number $d
     * @param        float $float
     */

    public function testToFloat($w, $n, $d, float $float)
    {
        // Given
        $number = new Rational($w, $n, $d);

        // When
        $computedFloat = $number->toFloat();

        // Then
