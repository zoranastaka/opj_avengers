    /**
     * Cumulative distribution function
     * https://en.wikipedia.org/wiki/Heaviside_step_function
     *
     *  |\+∞
     *  |   δ(x) dx = 1
     * \|-∞
     *
     * @param float $x
     * @todo how to handle x = 0, depending on context, some say CDF=.5 @ x=0
     *
     * @return int
     */

    public function cdf(float $x): int
    {
        if ($x >= 0) {
            return 1;
        }
        return 0;
    }
