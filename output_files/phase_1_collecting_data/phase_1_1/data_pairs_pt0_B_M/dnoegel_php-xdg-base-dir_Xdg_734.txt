    /**
     * @return string
     */

    public function getHomeConfigDir()
    {
        if ($path = getenv('XDG_CONFIG_HOME')) {
            return $path;
        }

        $homeDir = $this->getHomeDir();

        $path = DIRECTORY_SEPARATOR === $homeDir ? $homeDir.'.config' : $homeDir . DIRECTORY_SEPARATOR . '.config';

        return $path;
    }
