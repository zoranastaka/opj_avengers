    /**
     * @test         chiSquaredTest

     * @param        array $observed
     * @param        array $expected
     * @param        float $χ²
     * @param        float $p
     * @throws       \Exception
     */

    public function testChiSquaredTest(array $observed, array $expected, float $χ², float $p)
    {
        // When
        $chi = Significance::chiSquaredTest($observed, $expected);
