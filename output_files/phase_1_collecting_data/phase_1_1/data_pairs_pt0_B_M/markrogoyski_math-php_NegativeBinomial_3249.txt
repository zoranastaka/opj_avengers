    /**
     * Mode of the distribution
     *
     *        | p(r - 1) |
     * mode = | -------- |  for r > 1
     *        |_  1 - p _|
     *
     * mode = 0             for r â‰¤ 1
     *
     * @return float
     */

    public function mode(): float
    {
        if ($this->r <= 1) {
            return 0;
        }

        $r = $this->r;
        $p = $this->p;

        return \floor(($p * ($r - 1)) / (1 - $p));
    }
