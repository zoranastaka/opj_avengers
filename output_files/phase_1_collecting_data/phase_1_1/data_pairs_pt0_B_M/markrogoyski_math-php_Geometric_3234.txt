    /**
     * Median of the distribution
     *
     *           _           _
     *          |     -1      |
     * median = | ----------- | - 1
     *          | log₂(1 - p) |
     *
     * @return float
     */

    public function median(): float
    {
        $log₂⟮1 − p⟯ = \log(1 - $this->p, 2);

        return \ceil(-1 / $log₂⟮1 − p⟯) - 1;
    }
