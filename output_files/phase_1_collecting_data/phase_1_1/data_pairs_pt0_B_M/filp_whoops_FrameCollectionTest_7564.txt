    /**
     * @param  int   $total
     * @return array
     */

    public function getFrameDataList($total)
    {
        $total = max((int) $total, 1);
        $self  = $this;
        $frames = array_map(function () use ($self) {
            return $self->getFrameData();
        }, range(1, $total));

        return $frames;
    }
