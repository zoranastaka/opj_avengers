    /**
     * Raw data with each observation labeled
     * ['B1' => ['Heating value' => -0.1005049, 'C' => 0.6265746, 'H' => -1.1716630, ... ]]
     *
     * @return number[]
     */

    public function getLabeledYscData(): array
    {
        return \array_map(
            function (array $data) {
                return \array_combine(self::YSC_LABELS, $data);
            },
            self::YSC_DATA
        );
    }
