    /**
     * @test rand
     */

    public function testRand()
    {
        foreach (\range(-3, 3) as $μ) {
            foreach (\range(1, 3) as $σ) {
                // Given
                $log_normal = new LogNormal($μ, $σ);

                // When
                $random = $log_normal->rand();

                // Then
                $this->assertTrue(\is_numeric($random));
            }
        }
    }
