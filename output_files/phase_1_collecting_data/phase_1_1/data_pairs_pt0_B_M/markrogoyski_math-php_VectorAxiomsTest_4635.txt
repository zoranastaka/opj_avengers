    /**
     * @test Axiom: A⊥⋅A = 0
     * Perp dot product with itself will be zero.
     * @dataProvider dataProviderForPerpendicularIdentity
     */

    public function testPerpDotProductZero(array $A)
    {
        // Given
        $A = new Vector($A);

        // When
        $A⊥⋅A = $A->perpDotProduct($A);

        // Then
        $this->assertEquals(0, $A⊥⋅A);
    }
