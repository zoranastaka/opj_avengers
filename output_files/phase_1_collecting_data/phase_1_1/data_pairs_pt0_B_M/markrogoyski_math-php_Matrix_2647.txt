    /**
     * Walk a function over all elements of the matrix
     *
     * @param callable $func
     */

    public function walk(callable $func): void
    {
        $m = $this->m;
        $n = $this->n;

        for ($i = 0; $i < $m; $i++) {
            for ($j = 0; $j < $n; $j++) {
                $func($this->A[$i][$j]);
            }
        }
    }
