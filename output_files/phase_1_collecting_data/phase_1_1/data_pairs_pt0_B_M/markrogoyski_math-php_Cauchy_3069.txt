    /**
     * Inverse CDF (Quantile function)
     *
     * Q(p;x₀,γ) = x₀ + γ tan[π(p - ½)]
     *
     * @param float $p
     *
     * @return float
     */

    public function inverse(float $p): float
    {
        Support::checkLimits(['p' => '[0,1]'], ['p' => $p]);

        $x₀ = $this->x₀;
        $γ  = $this->γ;

        $π = \M_PI;

        return $x₀ + $γ * \tan($π * ($p - .5));
    }
