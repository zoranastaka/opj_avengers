    /**
     * @test rand
     */

    public function testRand()
    {
        foreach (\range(1, 10) as $k) {
            foreach (\range(1, 10) as $λ) {
                // Given
                $weibull = new Weibull($k, $λ);
                foreach (\range(1, 3) as $_) {
                    // When
                    $random = $weibull->rand();

                    // Then
                    $this->assertTrue(\is_numeric($random));
                }
            }
        }
    }
