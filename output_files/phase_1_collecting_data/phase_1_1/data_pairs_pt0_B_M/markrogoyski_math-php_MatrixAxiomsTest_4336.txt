    /**
     * @test Axiom: Jᵀ = J
     * Transpose of an exchange matrix is itself
     * @throws \Exception
     */

    public function testTransposeOfExchangeMatrix()
    {
        foreach (\range(1, 20) as $n) {
            // Given
            $J  = MatrixFactory::exchange($n);
            // When
            $Jᵀ = $J->transpose();
            // Then
            $this->assertEquals($J->getMatrix(), $Jᵀ->getMatrix());
        }
    }
