    /**
     * @test         det
     * @dataProvider dataProviderDet
     * @param        array $A
     * @param        Polynomial $expected
     */

    public function testDet(array $A, Polynomial $expected)
    {
        // Given
        $A = new ObjectSquareMatrix($A);

        $det = $A->det();

        // Then
        $this->assertEquals($det, $expected);

        // Then
