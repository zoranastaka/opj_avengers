    /**
     * @test         median
     * @dataProvider dataProviderForMedian
     * @param        number $λ
     * @param        number $expectedMedian
     */

    public function testMedian($λ, $expectedMedian)
    {
        // Given
        $exponential = new Exponential($λ);

        $this->assertEqualsWithDelta($expectedMedian, $median, 0.0000001);
