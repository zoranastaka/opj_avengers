    /**
     * Split current format to list of archivers.
     *
     * @param string $source
     * @return string[]|string
     */

    protected function _getArchivers($source)
    {
        $ext = pathinfo($source, PATHINFO_EXTENSION);
        if (!empty($this->_formats[$ext])) {
            return explode('.', $this->_formats[$ext]);
        }
        return [];
    }
