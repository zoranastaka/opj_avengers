    /**
     * @test   lehmerMean with a p of one is the arithmetic mean
     * @throws \Exception
     */

    public function testLehmerMeanPEqualsOneIsArithmeticMean()
    {

        $numbers = [ 3, 6, 2, 9, 1, 7, 2];

        $mean = Average::lehmerMean($numbers, $p);
