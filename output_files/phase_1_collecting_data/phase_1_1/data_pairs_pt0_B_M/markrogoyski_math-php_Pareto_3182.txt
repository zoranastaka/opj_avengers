    /**
     * Variance of the distribution
     *
     * σ² = ∞                 a ≤ 2
     *
     *            ab²
     * σ² = ---------------   a > 2
     *      (a - 1)²(a - 2)
     *
     * @return float
     */

    public function variance(): float
    {
        $a = $this->a;
        $b = $this->b;

        if ($a <= 2) {
            return \INF;
        }

        return ($a * $b ** 2) / (($a - 1) ** 2 * ($a - 2));
    }
