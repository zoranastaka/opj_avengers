    /**
     * Probability mass function
     *
     *       1
     * pmf = -
     *       n
     *
     * Percentile n = b - a + 1
     *
     * @return float
     */

    public function pmf(): float
    {
        $a = $this->a;
        $b = $this->b;

        $n = $b - $a + 1;

        return 1 / $n;
    }
