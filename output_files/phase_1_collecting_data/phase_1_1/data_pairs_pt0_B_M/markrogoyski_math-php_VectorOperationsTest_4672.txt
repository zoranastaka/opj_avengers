    /**
     * @test         subtract
     * @dataProvider dataProviderForSubtract
     */

    public function testSubtract(array $A, array $B, array $R)
    {
        // Given
        $A    = new Vector($A);
        $B    = new Vector($B);
        $R    = new Vector($R);

        // When
        $A−B = $A->subtract($B);

        // Then
        $this->assertEquals($R, $A−B);
        $this->assertEquals($R->getVector(), $A−B->getVector());
    }
