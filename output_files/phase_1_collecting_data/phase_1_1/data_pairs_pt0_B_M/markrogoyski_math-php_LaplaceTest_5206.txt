    /**
     * @test     mean is always μ
     */

    public function testMean()
    {
        foreach (\range(-5, 5) as $μ) {
            foreach (\range(1, 3) as $b) {
                // Given
                $laplace = new Laplace($μ, $b);

                // When
                $mean = $laplace->mean();

                // Then
                $this->assertEquals($μ, $mean);
            }
        }
    }
