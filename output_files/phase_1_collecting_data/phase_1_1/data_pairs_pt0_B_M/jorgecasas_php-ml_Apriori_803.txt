    /**
     * Generates frequent item sets.
     *
     * @return mixed[][][]
     */

    public function apriori(): array
    {
        $L = [];

        $items = $this->frequent($this->items());
        for ($k = 1; isset($items[0]); ++$k) {
            $L[$k] = $items;
            $items = $this->frequent($this->candidates($items));
        }

        return $L;
    }
