    /**
     * p-norm (|x|p)
     * Also known as lp norm
     *
     * https://en.wikipedia.org/wiki/Norm_(mathematics)#p-norm
     *
     * |x|p = (∑|xᵢ|ᵖ)¹/ᵖ
     *
     * @param number $p
     *
     * @return number
     */

    public function pNorm($p)
    {
        return \array_sum(Map\Single::pow(Map\Single::abs($this->A), $p)) ** (1 / $p);
    }
