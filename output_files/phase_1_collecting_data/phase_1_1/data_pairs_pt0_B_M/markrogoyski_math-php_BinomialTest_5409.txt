    /**
     * @test         cdf
     * @dataProvider dataProviderForCdf
     * @param        int $n
     * @param        int $r
     * @param        float $p
     * @param        float $expectedCdf
     */

    public function testCdf(int $n, int $r, float $p, float $expectedCdf)
    {
        // Given
        $binomial = new Binomial($n, $p);

        // When
        $cdf = $binomial->cdf($r);

        // Then
        $this->assertEqualsWithDelta($expectedCdf, $cdf, 0.0000001);
    }
