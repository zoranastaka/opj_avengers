    /** @test */

    public function it_hears_matching_commands_without_invokable_service()
    {
        $botman = $this->getBot([
            'sender' => 'UX12345',
            'recipient' => 'general',
            'message' => 'Foo',
        ]);
        $service = new InvokableService();
        $botman->hears('foo', $service);
        $botman->listen();
        $this->assertTrue($service->invoked);
