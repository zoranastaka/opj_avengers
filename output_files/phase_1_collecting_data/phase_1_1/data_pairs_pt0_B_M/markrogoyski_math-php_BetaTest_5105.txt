    /**
     * @test         cdf
     * @dataProvider dataProviderForCdf
     * @param        float $x
     * @param        float $α
     * @param        float $β
     * @param        float $expected_cdf
     */

    public function testCdf($x, $α, $β, $expected_cdf)
    {
        // Given
        $beta = new Beta($α, $β);

        // When
        $cdf = $beta->cdf($x);

        // Then
        $this->assertEqualsWithDelta($expected_cdf, $cdf, 0.000001);
    }
