    // optional trans: only return the string if it's translated

    function otrans($text)
    {
        $locale = Session::get(SESSION_LOCALE);

        if ($locale == 'en') {
            return trans($text);
        } else {
            $string = trans($text);
            $english = trans($text, [], 'en');

            return $string != $english ? $string : '';
        }
    }
