    /**
     * @test Axiom: A⋅A⊥ = 0
     * Vector dot product with a vector perpendicular to it will be zero.
     * @dataProvider dataProviderForPerpendicularIdentity
     */

    public function testPerpendicularDotProduct(array $A)
    {
        // Given
        $A = new Vector($A);

        // When
        $A⊥   = $A->perpendicular();
        $A⋅A⊥ = $A->dotProduct($A⊥);

        // Then
        $this->assertEquals(0, $A⋅A⊥);
    }
