    /**
     * @test         count
     * @dataProvider dataProviderForSingleSet
     * @param        array $A
     */

    public function testCount(array $A)
    {
        // Given
        $set = new Set($A);
        $expectedCount = count($A);

        // When
        $count = count($set);

        // Then
        $this->assertEquals($expectedCount, $count);
        $this->assertEquals($set->length(), $count);
    }
