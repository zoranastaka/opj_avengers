    /**
     * @test         pdf
     * @dataProvider dataProviderForPdf
     * @param        float $a
     * @param        float $b
     * @param        float $x
     * @param        float $expected
     * @throws       \Exception
     */

    public function testPdf(float $a, float $b, float $x, float $expected)

        // Given
        $uniform = new Uniform($a, $b);
