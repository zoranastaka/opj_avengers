    /** @test */

    public function it_sets_the_order_by_parameters()
    {
        $apiController = new ApiController;

        $apiController->setSortCriteria('created_at');

        $this->assertEquals(
            'created_at',
            $apiController->getSortCriteria()
        );

        $this->assertEquals(
            'asc',
            $apiController->getSortDirection()
        );

        $apiController->setSortCriteria('-created_at');

        $this->assertEquals(
            'created_at',
            $apiController->getSortCriteria()
        );

        $this->assertEquals(
            'desc',
            $apiController->getSortDirection()
        );
    }
