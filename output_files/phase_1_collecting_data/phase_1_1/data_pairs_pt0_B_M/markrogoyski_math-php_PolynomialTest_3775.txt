     * @param        array $polynomialA
     * @param        array $polynomialB
     * @param        array $expected_sum
     * @throws       \Exception
     */

    public function testAddition(array $polynomialA, array $polynomialB, array $expected_sum)
    {
        // Given
        $polynomialA  = new Polynomial($polynomialA);
