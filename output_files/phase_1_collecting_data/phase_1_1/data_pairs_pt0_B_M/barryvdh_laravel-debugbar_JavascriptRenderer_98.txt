    /**
     * Return assets as a string
     *
     * @param string $type 'js' or 'css'
     * @return string
     */

    public function dumpAssetsToString($type)
    {
        $files = $this->getAssets($type);

        $content = '';
        foreach ($files as $file) {
            $content .= file_get_contents($file) . "\n";
        }

        return $content;
    }
