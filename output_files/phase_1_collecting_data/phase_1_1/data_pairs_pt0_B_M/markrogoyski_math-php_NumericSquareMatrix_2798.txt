    /**
     * Constructor
     *
     * @param array $A
     *
     * @throws Exception\MathException
     */

    public function __construct(array $A)
    {
        parent::__construct($A);

        if ($this->m !== $this->n) {
            throw new Exception\MatrixException('Not a square matrix; row count and column count differ');
        }
    }
