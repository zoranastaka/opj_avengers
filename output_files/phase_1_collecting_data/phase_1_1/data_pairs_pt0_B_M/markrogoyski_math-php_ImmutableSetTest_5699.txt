    /**
     * @test addMulti does nothing
     */

    public function testAddMultiDoesNothing()
    {
        // Given
        $A = new ImmutableSet([1, 2, 3, 4]);
        $B = $A->copy();

        // When
        $A->addMulti([5]);
        $A->addMulti([6, 7, 8]);
        $A->addMulti([7, 8, 9]);
        $A->addMulti([new Set(['a', 'b'])]);

        // Then
        $this->assertEquals($B, $A);
        $this->assertEquals($B->asArray(), $A->asArray());
    }
