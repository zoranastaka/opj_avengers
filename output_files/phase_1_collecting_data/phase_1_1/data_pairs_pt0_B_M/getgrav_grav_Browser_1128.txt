    /**
     * Determine if the request comes from a human, or from a bot/crawler
     *
     * @return bool
     */

    public function isHuman()
    {
        $browser = $this->getBrowser();
        if (empty($browser)) {
            return false;
        }

        if (preg_match('~(bot|crawl)~i', $browser)) {
            return false;
        }

        return true;
    }
