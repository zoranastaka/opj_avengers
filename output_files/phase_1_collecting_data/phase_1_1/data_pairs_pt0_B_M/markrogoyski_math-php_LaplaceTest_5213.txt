    /**
     * @test rand
     */

    public function testRand()
    {
        foreach (\range(-3, 3) as $μ) {
            foreach (\range(1, 3) as $b) {
                // Given
                $laplace = new Laplace($μ, $b);

                // When
                $random = $laplace->rand();

                // Then
                $this->assertTrue(\is_numeric($random));
            }
        }
    }
