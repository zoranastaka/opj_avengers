    /**
     * @test Axiom: A⋅B = B⋅A
     * Dot product is commutative
     * @dataProvider dataProviderForTwoVectors
     */

    public function testDotProductCommutative(array $A, array $B)
    {
        // Given
        $A = new Vector($A);
        $B = new Vector($B);

        // When
        $A⋅B = $A->dotProduct($B);
        $B⋅A = $B->dotProduct($A);

        // Then
        $this->assertEquals($A⋅B, $B⋅A);
    }
