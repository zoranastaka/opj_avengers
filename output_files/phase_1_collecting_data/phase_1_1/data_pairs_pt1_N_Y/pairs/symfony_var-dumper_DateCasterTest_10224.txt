    /**
     * @dataProvider provideIntervals
     */

    public function testDumpInterval($intervalSpec, $ms, $invert, $expected)
    {
        $interval = $this->createInterval($intervalSpec, $ms, $invert);

        $xDump = <<<EODUMP
DateInterval {
  interval: $expected
%A}
EODUMP;

        $this->assertDumpMatchesFormat($xDump, $interval);
    }
