    /**
     * Merge current rules config with provided list of rules.
     *
     * @param array $rules
     * @return $this
     * @see setRules()
     * @see ArrayHelper::merge()
     */

    public function mergeRules(array $rules)
    {
        $this->setRules(ArrayHelper::merge($this->getRules(), $rules));

        return $this;
    }
