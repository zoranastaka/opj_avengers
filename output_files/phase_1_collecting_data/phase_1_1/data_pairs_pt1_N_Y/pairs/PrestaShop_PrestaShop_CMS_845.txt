    /**
     * @param $idCategory
     *
     * @return false|string|null
     */

    public static function getLastPosition($idCategory)
    {
        $sql = '
		SELECT MAX(position) + 1
		FROM `' . _DB_PREFIX_ . 'cms`
		WHERE `id_cms_category` = ' . (int) $idCategory;

        return Db::getInstance()->getValue($sql);
    }
