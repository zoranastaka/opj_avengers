    /**
     * {@inheritdoc}
     */

    protected function resetProperties(): void
    {
        $characters = preg_quote($this->delimiter, '/').'|'.preg_quote($this->enclosure, '/');
        $this->rfc4180_regexp = '/[\s|'.$characters.']/x';
        $this->rfc4180_enclosure = $this->enclosure.$this->enclosure;
    }
