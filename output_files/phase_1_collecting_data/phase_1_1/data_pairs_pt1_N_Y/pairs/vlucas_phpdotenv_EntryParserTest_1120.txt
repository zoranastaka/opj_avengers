    /**
     * @param \GrahamCampbell\ResultType\Result<\Dotenv\Parser\Entry,string> $result
     * @param string                                                         $error
     *
     * @return void
     */

    private function checkErrorResult(Result $result, string $error)

        self::assertSame($error, $result->error()->get());
