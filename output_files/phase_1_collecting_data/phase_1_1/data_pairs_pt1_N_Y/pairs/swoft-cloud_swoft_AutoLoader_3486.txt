/**
 * Class AutoLoader
 *
 * @since 2.0
 */

class AutoLoader extends SwoftComponent
{
    /**
     * @return array
     */
    public function getPrefixDirs(): array
    {
        return [
            __NAMESPACE__ => __DIR__,
        ];
    }

    /**
     * @return array
     */
    protected function metadata(): array
    {
        return [];
    }
}
