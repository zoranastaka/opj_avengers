    /**
     * Predict a single sample and return the result.
     *
     * @internal
     *
     * @param list<string|int|float> $sample
     * @return int|float
     */

    public function predictSample(array $sample)
    {
        /** @var \Rubix\ML\Graph\Nodes\Average $node */
        $node = $this->search($sample);

        return $node->outcome();
    }
