	/**
	 * Retrieve the file content in one go
	 * Once you retrieve the content you cannot use MimeMailParser_attachment::read()
	 * @return String
	 */

	public function getContent() {
		if ($this->content === null) {
			fseek($this->stream, 0);
			while(($buf = $this->read()) !== false) { 
				$this->content .= $buf; 
			}
		}
