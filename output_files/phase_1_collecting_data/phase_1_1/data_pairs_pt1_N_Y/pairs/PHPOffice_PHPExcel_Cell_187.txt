    /**
     * Get value binder to use
     *
     * @return PHPExcel_Cell_IValueBinder
     */

    public static function getValueBinder()
    {
        if (self::$valueBinder === null) {
            self::$valueBinder = new PHPExcel_Cell_DefaultValueBinder();
        }

        return self::$valueBinder;
    }
