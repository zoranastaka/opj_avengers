    /**
     * {@inheritdoc}
     *
     * This method only sends the content once.
     *
     * @return $this
     */

    public function sendContent()
    {
        if ($this->streamed) {
            return $this;
        }

        $this->streamed = true;

        if (null === $this->callback) {
            throw new \LogicException('The Response callback must not be null.');
        }

        ($this->callback)();

        return $this;
    }
