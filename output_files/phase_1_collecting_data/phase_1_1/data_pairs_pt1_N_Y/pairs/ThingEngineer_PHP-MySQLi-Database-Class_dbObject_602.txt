	/**
	 * chained method that append a field or fields to skipping
	 * @param mixed|array|false $field field name; array of names; empty skipping if false
	 * @return $this
	 */

    public function skip($field){
	    if(is_array($field)) {
		    foreach ($field as $f) {
			    $this->toSkip[] = $f;
		    }
	    } else if($field === false) {
	    	$this->toSkip = array();
	    } else{
	    	$this->toSkip[] = $field;
	    }
