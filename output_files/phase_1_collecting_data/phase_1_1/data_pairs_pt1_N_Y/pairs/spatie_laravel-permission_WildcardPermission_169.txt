    /**
     * @param Collection $part
     * @param Collection $otherPart
     *
     * @return bool
     */

    protected function containsAll(Collection $part, Collection $otherPart): bool
    {
        foreach ($otherPart->toArray() as $item) {
            if (! $part->contains($item)) {
                return false;
            }
        }

        return true;
    }
