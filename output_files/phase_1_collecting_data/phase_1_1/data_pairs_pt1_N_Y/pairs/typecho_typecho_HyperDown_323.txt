    /**
     * @param $url
     * @return string
     */

    public function cleanUrl($url)
    {
        if (preg_match("/^\s*((http|https|ftp|mailto):\S+)/i", $url, $matches)) {
            return $matches[1];
        } elseif (preg_match("/^\s*(\S+)/i", $url, $matches)) {
            return $matches[1];
        } else {
            return '#';
        }
    }
