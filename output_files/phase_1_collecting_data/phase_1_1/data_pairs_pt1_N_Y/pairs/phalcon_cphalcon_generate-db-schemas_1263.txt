/**
 * @param string $root
 *
 * @return array
 */

function getMigrations(string $root): array
{
    $path       = $root . '/tests/_data/fixtures/Migrations/';
    $migrations = [];
    foreach (glob($path . '*.php') as $file) {
        $file = str_replace([$path, '.php'], '', $file);
        if ($file !== 'AbstractMigration') {
            $migrations[] = $file;
        }
    }

    asort($migrations);

    return $migrations;
}
