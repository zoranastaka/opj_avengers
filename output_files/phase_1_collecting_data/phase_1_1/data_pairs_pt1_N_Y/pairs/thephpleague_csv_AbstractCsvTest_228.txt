    /**
     * @covers ::getDelimiter
     * @covers ::setDelimiter
     */

    public function testDelimiter(): void
    {
        $this->csv->setDelimiter('o');
        self::assertSame('o', $this->csv->getDelimiter());

        $this->expectException(InvalidArgument::class);

        $this->csv->setDelimiter('foo');
    }
