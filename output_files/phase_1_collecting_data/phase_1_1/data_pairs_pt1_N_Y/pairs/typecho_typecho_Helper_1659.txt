    /**
     * 导入语言项
     *
     * @access public
     * @param string $domain
     * @return void
     */

    public static function lang($domain)
    {
        $currentLang = Typecho_I18n::getLang();
        if ($currentLang) {
            $currentLang = basename($currentLang);
            $fileName = dirname(__FILE__) . '/' . $domain . '/lang/' . $currentLang;
            if (file_exists($fileName)) {
                Typecho_I18n::addLang($fileName);
            }
        }
    }
