    /**
     * @param $url
     * @param $file
     * @throws \Exception
     */

    public function download($url, $file)
    {
        if (!$url) {
            throw new \RuntimeException('Package url is missing.');
        }

        if (!file_put_contents($file, @fopen($url, 'r'))) {
            App::abort(500, 'Download failed or path not writable.');
        }
    }
